5. COMMAND PROCESSOR
public async Task Process(string phone, string text, string groupId)
{
    var user = _db.WhatsAppUsers.Find(phone);
    var group = _db.WhatsAppGroups.Find(groupId);

    if (user == null || group == null) return;

    if (user.Role != group.SystemRole) return;

    var parts = text.Split(' ');
    var cmd = parts[0].ToUpper();
    var refNo = parts.Length > 1 ? parts[1] : null;

    switch (cmd)
    {
        case "APPROVE": await ApproveVoucher(refNo, phone); break;
        case "REJECT": await RejectVoucher(refNo, phone, text); break;
        case "PAY": await PayVoucher(refNo, phone); break;
        case "STATUS": await SendVoucherStatus(refNo, groupId); break;
    }
}
